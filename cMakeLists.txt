#Minimal cMakeList for SDL2
cmake_minimum_required(VERSION 3.8)

project(AABB)

set(CMAKE_C_STANDARD 11)

set(SOURCE_FILES main.c aabb.c)

add_executable(AABB ${SOURCE_FILES})

INCLUDE(FindPkgConfig)
PKG_SEARCH_MODULE(SDL2 REQUIRED sdl2)
PKG_SEARCH_MODULE(SDL2_TTF required sdl2_ttf)

INCLUDE_DIRECTORIES(${SDL2_INCLUDE_DIRS} ${SDL2_TTF_INCLUDE_DIRS} ${include})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${SDL2_LIBRARIES} ${SDL2_TTF_LIBRARIES})
